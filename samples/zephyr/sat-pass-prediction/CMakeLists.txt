cmake_minimum_required(VERSION 3.13.1)
find_package(Zephyr REQUIRED HINTS $ENV{ZEPHYR_BASE})

project(app LANGUAGES C)

# Pass prediction needs the current Unix time, but this sample uses the device
# uptime counter source for crypto (no UTC provisioning). We bridge the gap by
# capturing the build's wall-clock time here and injecting it as a compile
# definition; main.c computes "now" as SAMPLE_BUILD_UNIX_TIME + uptime.
#
# Captured at CMake *configure* time, so build with `-p always` right before
# flashing to keep the timestamp fresh.
string(TIMESTAMP SAMPLE_BUILD_UNIX_TIME "%s" UTC)
target_compile_definitions(app PRIVATE SAMPLE_BUILD_UNIX_TIME=${SAMPLE_BUILD_UNIX_TIME})

# The Hubble master key (base64) and device location are provisioned at build
# time via Kconfig, e.g.
#   west build -p always -b <board> . -- \
#       -DCONFIG_SAMPLE_HUBBLE_KEY="<base64 key>" \
#       -DCONFIG_SAMPLE_DEVICE_LAT="37.7749" -DCONFIG_SAMPLE_DEVICE_LON="-122.4194"
#
# The sats[] table is generated by tools/orbital_params_fetch.py into
# src/sat_params.generated.c, which is gitignored so regenerating never dirties
# the tree. Prefer it when present; otherwise fall back to the committed all-zero
# placeholder (src/sat_params.c) so build-only CI still passes. The chosen file
# is #included by satellites.c (which derives the element count), so it is not
# listed as a source here.
set(sat_params_file "${CMAKE_CURRENT_SOURCE_DIR}/src/sat_params.c")
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/src/sat_params.generated.c")
  set(sat_params_file "${CMAKE_CURRENT_SOURCE_DIR}/src/sat_params.generated.c")
endif()
target_compile_definitions(app PRIVATE SAT_PARAMS_FILE="${sat_params_file}")

target_sources(app PRIVATE src/main.c src/satellites.c)
